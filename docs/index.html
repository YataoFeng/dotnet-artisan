Key Principles

KISS First

Simple CRUD doesn't need DDD, MediatR, or CQRS. Scale to the problem.

No Repository Wrapping

Use DbContext directly. EF Core IS the repository and unit of work.

Version-Aware

Guidance adapts to net8.0 → net9.0 → net10.0 → net11.0 automatically.

Framework First

TimeProvider, ILogger<T>, IHttpClientFactory, System.Text.Json — use what's built in.

Bad vs Good — Code Patterns

Anti-patterns caught in code review. Click any card to expand.

DI Lifetime Mismatch ▼

BAD — Scoped service injected into Singleton
builder.Services.AddSingleton<ICache, MyCache>();
builder.Services.AddScoped<AppDbContext>();

// MyCache depends on DbContext — but Singleton wraps Scoped!
public class MyCache(AppDbContext db) : ICache { }
GOOD — Match lifetimes correctly
builder.Services.AddSingleton<ICache, MyCache>();
builder.Services.AddScoped<AppDbContext>();

// Use IServiceScopeFactory to resolve scoped deps in singletons
public class MyCache(IServiceScopeFactory sf) : ICache
{
    public async Task<T> Get<T>(string key, Func<AppDbContext, Task<T>> factory)
    {
        using var scope = sf.CreateScope();
        var db = scope.ServiceProvider.GetRequiredService<AppDbContext>();
        return await factory(db);
    }
}

Async Void Fire-and-Forget ▼

BAD — Exceptions swallowed, no tracking
public void SendEmail(string to, string body)
{
    _ = Task.Run(async () =>
    {
        await _emailService.SendAsync(to, body);
    }); // Exception? What exception?
}
GOOD — Use BackgroundService or proper fire-and-forget
// Option A: BackgroundService with Channel
public class EmailBackgroundService(Channel<EmailJob> queue) : BackgroundService
{
    protected override async Task ExecuteAsync(CancellationToken ct)
    {
        await foreach (var job in queue.Reader.ReadAllAsync(ct))
            await _emailService.SendAsync(job.To, job.Body, ct);
    }
}

// Option B: Proper fire-and-forget with exception handling
public static void FireAndForget(this Task task, ILogger logger)
{
    task.ContinueWith(t => logger.LogError(t.Exception, "Unhandled bg error"),
        TaskContinuationOptions.OnlyOnFaulted);
}

DateTime.Now vs TimeProvider ▼

BAD — Not testable, timezone-dependent
public class OrderService(AppDbContext db)
{
    public async Task CreateOrder(CreateOrderRequest req)
    {
        var order = new Order
        {
            CreatedAt = DateTime.Now, // Can't test time-based logic!
            ExpiresAt = DateTime.UtcNow.AddHours(24)
        };
        db.Orders.Add(order);
        await db.SaveChangesAsync();
    }
}
GOOD — TimeProvider is injectable and testable
public class OrderService(AppDbContext db, TimeProvider clock)
{
    public async Task CreateOrder(CreateOrderRequest req)
    {
        var order = new Order
        {
            CreatedAt = clock.GetLocalNow().DateTime,
            ExpiresAt = clock.GetUtcNow().AddHours(24)
        };
        db.Orders.Add(order);
        await db.SaveChangesAsync();
    }
}

// Test: TimeProvider.Fake("2026-03-15T10:00:00Z")

N+1 EF Core Queries ▼

BAD — One query per order to get items
var orders = await db.Orders.Where(o => o.CustomerId == id).ToListAsync();
foreach (var order in orders) // N+1!
{
    var items = await db.OrderItems.Where(i => i.OrderId == order.Id).ToListAsync();
    order.Items = items;
}
GOOD — Single query with Include
var orders = await db.Orders
    .Include(o => o.Items)
    .Where(o => o.CustomerId == id)
    .AsNoTracking()
    .ToListAsync();

Repository over DbContext ▼

BAD — Useless abstraction over EF Core
public interface IRepository<T> { Task<T> GetById(int id); Task Add(T e); Task Save(); }
public class Repository<T>(AppDbContext db) : IRepository<T> { /* wraps DbSet */ }
public class OrderService(IRepository<Order> repo) { /* uses repo */ }

// DbContext is the repository. DbSet<T> is the repository.
// This wrapper adds zero value, hides IQueryable, and blocks .Include().
GOOD — Use DbContext directly
public class OrderService(AppDbContext db)
{
    public async Task<Order?> Get(int id) => db.Orders.FindAsync(id);
    public async Task Add(Order o) { db.Orders.Add(o); await db.SaveChangesAsync(); }
    public IQueryable<Order> Query() => db.Orders.AsNoTracking(); // let caller compose
}

Missing CancellationToken ▼

BAD — No cancellation support
public async Task<List<Product>> Search(string q)
{
    return await db.Products.Where(p => p.Name.Contains(q)).ToListAsync();
    // Can't cancel this. Request timeout = stuck thread.
}
GOOD — Propagate CancellationToken end-to-end
public async Task<List<Product>> Search(string q, CancellationToken ct = default)
{
    return await db.Products.Where(p => p.Name.Contains(q)).ToListAsync(ct);
}
// ASP.NET Core binds HttpContext.RequestAborted to ct automatically
